CREATE TABLE STUDENT (
     STU_NUMBER		VARCHAR(12) PRIMARY KEY
     ,STU_NAME		VARCHAR(10) NOT NULL
     ,STU_BIRTHDAY	VARCHAR(10) NOT NULL
     ,STU_PHONE		VARCHAR(15) NOT NULL
     ,STU_EMAIL		VARCHAR(50) NOT NULL
     ,STU_ENTERANCE	INT(2) 		NOT NULL
     ,STU_AUTHORITY	INT(2) 		NOT NULL
     ,STU_GENDER		INT(4) 		NOT NULL
     ,STU_PASSWORD	VARCHAR(20) NOT NULL
     ,STU_DELETE_YN	VARCHAR(2) 	NOT NULL	DEFAULT 'N' 
);

CREATE TABLE LEDGER (
     LEDG_NUMBER			INT(10) 		PRIMARY KEY
     ,LEDG_DATE				VARCHAR(10) 	NOT NULL
     ,LEDG_CONTENT			VARCHAR(100) 	NOT NULL
     ,LEDG_TRANSE_TYPE		VARCHAR(10) 	NOT NULL
     ,LEDG_TRADE_PARTNER	VARCHAR(50) 	NOT NULL
     ,LEDG_METHOD			INT(2) 			NOT NULL
     ,LEDG_CATEGORY			VARCHAR(10) 	NOT NULL
     ,LEDG_BALANCE			INT(20) 		NOT NULL
     ,LEDG_REMARK			VARCHAR(100) 	NOT NULL
     ,LEDG_AMOUNT			INT(20) 		NOT NULL
     ,LEDG_FILE				VARCHAR(20) 	NOT NULL
 );
 
CREATE TABLE LOG_DELETE(
  	 STU_NUMBER	VARCHAR(12) PRIMARY KEY
     ,LOG_REASON	VARCHAR(100)
	 );
     
CREATE TABLE EXPERIENCE (
     STU_NUMBER 	VARCHAR(12) 	NOT NULL
     ,EXP_YEAR 		INT(4) 			NOT NULL
     ,EXP_SEMESTER 	INT(1) 			NOT NULL
     ,EXP_CONTENT 	VARCHAR(100) 	NOT NULL
     ,EXP_DELETE_YN VARCHAR(2) 		NOT NULL DEFAULT 'N' 
     );
     
CREATE TABLE STUDENT_FEE (
     STU_NUMBER 		VARCHAR(12) PRIMARY KEY
     ,FEE_PAID_DATE 	VARCHAR(10) DEFAULT '00000000' 	NOT NULL 
     ,FEE_TOTAL_AMOUNT 	INT(6) 		DEFAULT 0 			NOT NULL 
     ,FEE_PAID_AMOUNT 	INT(6) 		DEFAULT 0 			NOT NULL 
     ,FEE_PAID_METHOD 	INT(1) 		DEFAULT 1 			NOT NULL 
     ,FEE_PAID_STATUS 	INT(1) 		DEFAULT 3 			NOT NULL 
     ,FEE_CONTENT 		VARCHAR(150)
     ,FEE_DELETE_YN 	VARCHAR(2) 	DEFAULT 'N' 		NOT NULL
     );

CREATE TABLE COMMON (
     COMMON_CATEGORY	VARCHAR(20) NOT NULL
     ,COMMON_CODE 		INT(2)		NOT NULL
     ,COMMON_VALUE		VARCHAR(20) NOT NULL
     );
